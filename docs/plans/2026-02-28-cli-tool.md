# CLI Tool Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Add a unified CLI (`llm-sizer`) that exposes the calculation engine, data queries, docker/serve command generation, profiling config generation, and benchmark analysis — all sharing the same data and formulas as the web UI.

**Architecture:** Unified `commander` CLI entry point at `src/cli/index.ts` with subcommands. Pure calculation functions from `src/utils/` are imported directly (zero React deps). Data loaded from `src/data/*.json` via a shared loader. Output auto-detects TTY (table for interactive, JSON for pipes). Built separately from the web UI using `tsup`.

**Tech Stack:** TypeScript, commander (already a devDep), chalk (already a devDep), tsup (new devDep), tsx (already a devDep for dev mode), jest (existing test runner)

**PRD:** `docs/plans/cli-tool-prd.md`

// Resolve data directory relative to this file's location

function dataDir(): string {
  // Walk up from src/cli/utils/ to project root, then into src/data/
  return join(__dirname, '..', '..', 'data');

import type { Model, GPU, EmbeddingModel, RerankingModel, ModelType } from '../../types';
import { loadModels, loadGPUs, loadEmbeddingModels, loadRerankingModels } from './data-loader';

export function resolveModel(id: string): Model | null {
  const models = loadModels();
  return models.find(m => m.id === id) ?? null;
}

export function resolveGPU(id: string): GPU | null {
  const gpus = loadGPUs();
  return gpus.find(g => g.id === id) ?? null;
}

export function resolveModelByType(
  id: string,
  type: ModelType
): Model | EmbeddingModel | RerankingModel | null {
  switch (type) {
    case 'generation':
      return resolveModel(id);
    case 'embedding': {
      const models = loadEmbeddingModels();
      return models.find(m => m.id === id) ?? null;
    }
    case 'reranking': {
      const models = loadRerankingModels();
      return models.find(m => m.id === id) ?? null;
    }
    default:
      return null;
  }
}

/**
 * Resolve a model ID to its HuggingFace model ID.
 * Returns the hf_model_id field from models.json, or null if not found.
 */
export function resolveHFModelId(id: string): string | null {
  const model = resolveModel(id);
  return model?.hf_model_id ?? null;
}
```

export type OutputFormat = 'json' | 'table';

/**
 * Detect the appropriate output format.
 * If explicitly specified, use that. Otherwise default to 'table'.
 * Callers can check process.stdout.isTTY to decide the default.
 */
export function detectOutputFormat(explicit?: string): OutputFormat {
  if (explicit === 'json') return 'json';
  if (explicit === 'table') return 'table';
  return 'table';
}

/**
 * Format data for output. JSON mode returns pretty-printed JSON.
 * Table mode returns a simple key-value representation (commands
 * will implement their own richer table formatting).
 */
export function formatOutput(data: Record<string, unknown>, format: OutputFormat): string {
  if (format === 'json') {
    return JSON.stringify(data, null, 2);
  }

  // Simple key-value table (individual commands override with richer formatting)
  const lines: string[] = [];
  for (const [key, value] of Object.entries(data)) {
    if (typeof value === 'object' && value !== null) {
      lines.push(`${key}:`);
      for (const [subKey, subValue] of Object.entries(value as Record<string, unknown>)) {
        lines.push(`  ${subKey}: ${subValue}`);
      }
    } else {
      lines.push(`${key}: ${value}`);
    }
  }
  return lines.join('\n');
}
```

Create `src/cli/commands/calculate.ts`:
```typescript
import { Command } from 'commander';
import chalk from 'chalk';
import type {
  InferenceQuantization,
  KVCacheQuantization,
  ModelType,
  CalculationResults,
} from '../../types';
import { calculateMemoryRequirements } from '../../utils/memoryCalculator';
import { estimatePerformance } from '../../utils/performanceEstimator';
import { resolveModel, resolveGPU, resolveModelByType } from '../utils/model-resolver';
import { formatOutput, detectOutputFormat } from '../utils/output';

export interface CalculateOptions {
  model: string;
  gpu: string;
  gpus: number;
  quant: InferenceQuantization;
  kvQuant: KVCacheQuantization;
  type: ModelType;
  batchSize: number;
  seqLength: number;
  users: number;
  output?: string;
}

export interface CalculateResult extends CalculationResults {
  inputs: {
    model: string;
    gpu: string;
    numGPUs: number;
    quantization: string;
    kvCacheQuantization: string;
    batchSize: number;
    sequenceLength: number;
    concurrentUsers: number;
  };
}

export function buildCalculateResult(opts: CalculateOptions): CalculateResult {
  const model = resolveModel(opts.model);
  if (!model) {
    throw new Error(`Model not found: ${opts.model}`);
  }

  const gpu = resolveGPU(opts.gpu);
  if (!gpu) {
    throw new Error(`GPU not found: ${opts.gpu}`);
  }

  const memoryResults = calculateMemoryRequirements(
    model,
    gpu,
    opts.quant,
    opts.kvQuant,
    opts.batchSize,
    opts.seqLength,
    opts.users,
    opts.gpus,
  );

  const performanceMetrics = estimatePerformance(
    model,
    gpu,
    opts.quant,
    opts.batchSize,
    opts.seqLength,
    opts.users,
    opts.gpus,
    memoryResults.vramPercentage,
  );

  return {
    ...memoryResults,
    performance: performanceMetrics,
    inputs: {
      model: opts.model,
      gpu: opts.gpu,
      numGPUs: opts.gpus,
      quantization: opts.quant,
      kvCacheQuantization: opts.kvQuant,
      batchSize: opts.batchSize,
      sequenceLength: opts.seqLength,
      concurrentUsers: opts.users,
    },
  };
}

function formatTable(result: CalculateResult): string {
  const lines: string[] = [];
  const modelName = resolveModel(result.inputs.model)?.name ?? result.inputs.model;
  const gpuName = resolveGPU(result.inputs.gpu)?.name ?? result.inputs.gpu;
  const gpuCount = result.inputs.numGPUs;

  lines.push(`Model: ${modelName} | GPU: ${gpuCount}x ${gpuName} | Quant: ${result.inputs.quantization.toUpperCase()}`);
  lines.push('');

  const statusIcon = result.status === 'okay' ? chalk.green('[OK]')
    : result.status === 'warning' ? chalk.yellow('[WARN]')
    : chalk.red('[ERROR]');

  lines.push(`Memory Usage: ${result.usedVRAM.toFixed(1)} / ${result.totalVRAM.toFixed(1)} GB (${result.vramPercentage.toFixed(1)}%)  ${statusIcon}`);
  lines.push('');

  const breakdown = result.memoryBreakdown;
  const total = result.usedVRAM;
  const maxBar = 30;

  const components: Array<[string, number]> = [
    ['Base Weights', breakdown.baseWeights],
    ['KV Cache', breakdown.kvCache],
    ['Activations', breakdown.activations],
    ['Framework OH', breakdown.frameworkOverhead],
    ['Multi-GPU OH', breakdown.multiGPUOverhead],
  ];

  for (const [name, value] of components) {
    if (value <= 0) continue;
    const barLen = Math.max(1, Math.round((value / total) * maxBar));
    const bar = '\u2588'.repeat(barLen);
    lines.push(`  ${name.padEnd(16)} ${value.toFixed(1).padStart(6)} GB  ${bar}`);
  }

  if (result.performance.totalThroughput > 0) {
    lines.push('');
    lines.push(`Performance: ${result.performance.totalThroughput} tok/s total | ${result.performance.perUserSpeed} tok/s per user`);
  }

  if (result.message) {
    lines.push('');
    lines.push(result.message);
  }

  return lines.join('\n');
}

export function registerCalculateCommand(program: Command): void {
  program
    .command('calculate')
    .description('Calculate memory requirements and performance estimates')
    .requiredOption('-m, --model <id>', 'Model ID from database')
    .requiredOption('-g, --gpu <id>', 'GPU ID from database')
    .option('-n, --gpus <count>', 'Number of GPUs', parseInt, 1)
    .option('--quant <type>', 'Inference quantization (fp16|fp8|int8|int4)', 'fp16')
    .option('--kv-quant <type>', 'KV cache quantization (fp16_bf16|fp8_bf16|int8)', 'fp16_bf16')
    .option('--type <type>', 'Model type (generation|embedding|reranking)', 'generation')
    .option('--batch-size <n>', 'Batch size', parseInt, 1)
    .option('--seq-length <n>', 'Sequence length', parseInt, 4096)
    .option('--users <n>', 'Concurrent users', parseInt, 1)
    .option('-o, --output <format>', 'Output format (json|table)')
    .action((opts) => {
      try {
        const result = buildCalculateResult({
          model: opts.model,
          gpu: opts.gpu,
          gpus: opts.gpus,
          quant: opts.quant as InferenceQuantization,
          kvQuant: opts.kvQuant as KVCacheQuantization,
          type: opts.type as ModelType,
          batchSize: opts.batchSize,
          seqLength: opts.seqLength,
          users: opts.users,
        });

        const format = detectOutputFormat(opts.output);
        if (format === 'json') {
          console.log(formatOutput(result as unknown as Record<string, unknown>, 'json'));
        } else {
          console.log(formatTable(result));
        }
      } catch (err: unknown) {
        const message = err instanceof Error ? err.message : String(err);
        console.error(chalk.red(`Error: ${message}`));
        process.exit(1);
      }
    });
}
```

import { Command } from 'commander';
import chalk from 'chalk';
import type { Model, EmbeddingModel, RerankingModel, ModelType } from '../../types';
import { loadModels, loadEmbeddingModels, loadRerankingModels } from '../utils/data-loader';
import { formatOutput, detectOutputFormat } from '../utils/output';

interface ListOptions {
  type?: ModelType;
  filter?: string;
  arch?: string;
  modality?: string;
}

export function listModels(opts: ListOptions): Array<Model | EmbeddingModel | RerankingModel> {
  const type = opts.type ?? 'generation';

  let models: Array<Model | EmbeddingModel | RerankingModel>;

  switch (type) {
    case 'embedding':
      models = loadEmbeddingModels();
      break;
    case 'reranking':
      models = loadRerankingModels();
      break;
    default:
      models = loadModels();
      break;
  }

  if (opts.filter) {
    const filterLower = opts.filter.toLowerCase();
    models = models.filter(
      m => m.id.toLowerCase().includes(filterLower) || m.name.toLowerCase().includes(filterLower)
    );
  }

  if (opts.arch && type === 'generation') {
    models = (models as Model[]).filter(m => m.architecture === opts.arch);
  }

  if (opts.modality && type === 'generation') {
    models = (models as Model[]).filter(m => (m.modality ?? 'text') === opts.modality);
  }

  return models;
}

export function showModel(
  id: string,
  type: ModelType
): Model | EmbeddingModel | RerankingModel | null {
  const models = listModels({ type });
  return models.find(m => m.id === id) ?? null;
}

function formatModelTable(models: Array<Model | EmbeddingModel | RerankingModel>, type: ModelType): string {
  if (models.length === 0) return 'No models found.';

  const lines: string[] = [];

  if (type === 'generation') {
    lines.push(`${'ID'.padEnd(35)} ${'Name'.padEnd(30)} ${'Params'.padStart(8)} ${'Arch'.padEnd(12)} ${'Context'.padStart(8)}`);
    lines.push('-'.repeat(97));
    for (const m of models as Model[]) {
      lines.push(
        `${m.id.padEnd(35)} ${m.name.padEnd(30)} ${(m.parameters_billions + 'B').padStart(8)} ${m.architecture.padEnd(12)} ${m.default_context_length.toString().padStart(8)}`
      );
    }
  } else if (type === 'embedding') {
    lines.push(`${'ID'.padEnd(30)} ${'Name'.padEnd(30)} ${'Params'.padStart(10)} ${'Dims'.padStart(6)} ${'MaxTok'.padStart(8)}`);
    lines.push('-'.repeat(88));
    for (const m of models as EmbeddingModel[]) {
      lines.push(
        `${m.id.padEnd(30)} ${m.name.padEnd(30)} ${(m.parameters_millions + 'M').padStart(10)} ${m.dimensions.toString().padStart(6)} ${m.max_tokens.toString().padStart(8)}`
      );
    }
  } else {
    lines.push(`${'ID'.padEnd(30)} ${'Name'.padEnd(30)} ${'Params'.padStart(10)} ${'MaxQuery'.padStart(10)} ${'MaxDoc'.padStart(8)}`);
    lines.push('-'.repeat(92));
    for (const m of models as RerankingModel[]) {
      lines.push(
        `${m.id.padEnd(30)} ${m.name.padEnd(30)} ${(m.parameters_millions + 'M').padStart(10)} ${m.max_query_length.toString().padStart(10)} ${m.max_doc_length.toString().padStart(8)}`
      );
    }
  }

  lines.push('');
  lines.push(`${models.length} model(s) found`);
  return lines.join('\n');
}

export function registerModelsCommand(program: Command): void {
  const cmd = program
    .command('models')
    .description('Query the model database');

  cmd
    .command('list')
    .description('List models')
    .option('--type <type>', 'Model type (generation|embedding|reranking)', 'generation')
    .option('--filter <text>', 'Filter by name/ID substring')
    .option('--arch <arch>', 'Filter by architecture (transformer|moe)')
    .option('--modality <mod>', 'Filter by modality (text|multimodal)')
    .option('-o, --output <format>', 'Output format (json|table)')
    .action((opts) => {
      const models = listModels({
        type: opts.type as ModelType,
        filter: opts.filter,
        arch: opts.arch,
        modality: opts.modality,
      });

      const format = detectOutputFormat(opts.output);
      if (format === 'json') {
        console.log(JSON.stringify(models, null, 2));
      } else {
        console.log(formatModelTable(models, opts.type as ModelType));
      }
    });

  cmd
    .command('show <id>')
    .description('Show model details')
    .option('--type <type>', 'Model type (generation|embedding|reranking)', 'generation')
    .option('-o, --output <format>', 'Output format (json|table)')
    .action((id, opts) => {
      const model = showModel(id, opts.type as ModelType);
      if (!model) {
        console.error(chalk.red(`Model not found: ${id}`));
        process.exit(1);
      }

      const format = detectOutputFormat(opts.output);
      if (format === 'json') {
        console.log(JSON.stringify(model, null, 2));
      } else {
        console.log(formatOutput(model as unknown as Record<string, unknown>, 'table'));
      }
    });
}

import { Command } from 'commander';
import chalk from 'chalk';
import type { GPU } from '../../types';
import { loadGPUs } from '../utils/data-loader';
import { detectOutputFormat } from '../utils/output';

interface ListGPUOptions {
  vendor?: string;
  tier?: string;
  minVram?: number;
  partitioning?: boolean;
}

export function listGPUs(opts: ListGPUOptions): GPU[] {
  let gpus = loadGPUs();

  if (opts.vendor) {
    gpus = gpus.filter(g => g.vendor === opts.vendor);
  }
  if (opts.tier) {
    gpus = gpus.filter(g => g.tier === opts.tier);
  }
  if (opts.minVram) {
    gpus = gpus.filter(g => g.vram_gb >= opts.minVram!);
  }
  if (opts.partitioning) {
    gpus = gpus.filter(g => g.partitioning?.supported);
  }

  return gpus;
}

export function showGPU(id: string): GPU | null {
  const gpus = loadGPUs();
  return gpus.find(g => g.id === id) ?? null;
}

export function compareGPUs(ids: string[]): GPU[] {
  const gpus = loadGPUs();
  return ids.map(id => gpus.find(g => g.id === id)).filter((g): g is GPU => g != null);
}

function formatGPUTable(gpus: GPU[]): string {
  if (gpus.length === 0) return 'No GPUs found.';

  const lines: string[] = [];
  lines.push(`${'ID'.padEnd(20)} ${'Name'.padEnd(28)} ${'Vendor'.padEnd(8)} ${'VRAM'.padStart(8)} ${'BW (GB/s)'.padStart(10)} ${'FP16 TFLOPS'.padStart(12)} ${'Tier'.padEnd(12)}`);
  lines.push('-'.repeat(102));

  for (const g of gpus) {
    lines.push(
      `${g.id.padEnd(20)} ${g.name.padEnd(28)} ${g.vendor.padEnd(8)} ${(g.vram_gb + 'GB').padStart(8)} ${g.memory_bandwidth_gbps.toString().padStart(10)} ${g.compute_tflops_fp16.toString().padStart(12)} ${g.tier.padEnd(12)}`
    );
  }

  lines.push('');
  lines.push(`${gpus.length} GPU(s) found`);
  return lines.join('\n');
}

function formatGPUCompare(gpus: GPU[]): string {
  if (gpus.length === 0) return 'No GPUs found.';

  const fields: Array<[string, (g: GPU) => string]> = [
    ['Name', g => g.name],
    ['Vendor', g => g.vendor],
    ['Tier', g => g.tier],
    ['VRAM', g => `${g.vram_gb} GB`],
    ['Memory Type', g => g.memory_type],
    ['Bandwidth', g => `${g.memory_bandwidth_gbps} GB/s`],
    ['FP16 TFLOPS', g => g.compute_tflops_fp16.toString()],
    ['FP8 TFLOPS', g => g.compute_tflops_fp8?.toString() ?? 'N/A'],
    ['TDP', g => `${g.tdp_watts}W`],
    ['PCIe', g => `Gen ${g.pcie_gen}`],
    ['Multi-GPU', g => g.multi_gpu_capable ? 'Yes' : 'No'],
    ['Partitioning', g => g.partitioning?.supported ? 'Yes' : 'No'],
  ];

  const labelWidth = 14;
  const colWidth = 28;
  const lines: string[] = [];

  lines.push(''.padEnd(labelWidth) + gpus.map(g => g.id.padEnd(colWidth)).join(''));
  lines.push('-'.repeat(labelWidth + colWidth * gpus.length));

  for (const [label, getter] of fields) {
    lines.push(label.padEnd(labelWidth) + gpus.map(g => getter(g).padEnd(colWidth)).join(''));
  }

  return lines.join('\n');
}

export function registerGPUsCommand(program: Command): void {
  const cmd = program
    .command('gpus')
    .description('Query the GPU database');

  cmd
    .command('list')
    .description('List GPUs')
    .option('--vendor <vendor>', 'Filter by vendor (AMD|NVIDIA)')
    .option('--tier <tier>', 'Filter by tier (datacenter|professional|consumer)')
    .option('--min-vram <gb>', 'Minimum VRAM in GB', parseInt)
    .option('--partitioning', 'Only GPUs with partition support')
    .option('-o, --output <format>', 'Output format (json|table)')
    .action((opts) => {
      const gpus = listGPUs({
        vendor: opts.vendor,
        tier: opts.tier,
        minVram: opts.minVram,
        partitioning: opts.partitioning,
      });

      const format = detectOutputFormat(opts.output);
      if (format === 'json') {
        console.log(JSON.stringify(gpus, null, 2));
      } else {
        console.log(formatGPUTable(gpus));
      }
    });

  cmd
    .command('show <id>')
    .description('Show GPU details')
    .option('-o, --output <format>', 'Output format (json|table)')
    .action((id, opts) => {
      const gpu = showGPU(id);
      if (!gpu) {
        console.error(chalk.red(`GPU not found: ${id}`));
        process.exit(1);
      }

      const format = detectOutputFormat(opts.output);
      console.log(format === 'json' ? JSON.stringify(gpu, null, 2) : formatGPUCompare([gpu]));
    });

  cmd
    .command('compare <ids...>')
    .description('Compare GPUs side by side')
    .option('-o, --output <format>', 'Output format (json|table)')
    .action((ids, opts) => {
      const gpus = compareGPUs(ids);
      if (gpus.length === 0) {
        console.error(chalk.red('No valid GPU IDs provided.'));
        process.exit(1);
      }

      const format = detectOutputFormat(opts.output);
      console.log(format === 'json' ? JSON.stringify(gpus, null, 2) : formatGPUCompare(gpus));
    });
}
